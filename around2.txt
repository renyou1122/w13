def turn_right():
    turn_left()
    turn_left()
    turn_left()

# 走 n 步
def move_n(n):
    for _ in range(n):
        move()

# 起點 (1,4) 朝東

# 1. 向東走6格
move_n(6)

# 2. 向北走4格
turn_left()  # 朝北
move_n(4)

# 3. 向西走5格
turn_left()  # 朝西
move_n(5)

# 4. 向北走5格
turn_right() # 朝北
move_n(5)

# 5. 向西走4格
turn_left()  # 朝西
move_n(4)

# 6. 向南走9格
turn_left()  # 朝南
move_n(9)

# 7. 向東走3格，回到起點
turn_left()  # 朝東
move_n(3)
